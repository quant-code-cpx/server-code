## 三、Phase 1：事件影响分析（Event Study）

### 3.1 核心算法：事件窗口超额收益

```
           估计窗口                  事件窗口
  ────────────────────────  ─────────────────────────
  [-estimation_end, -pre]   [-pre, 0, +post]
                              │
                            事件日 (T=0)

超额收益 AR(t)    = R_stock(t) - R_benchmark(t)
累计超额收益 CAR  = Σ AR(t)，t ∈ [-pre, +post]
平均超额收益 AAR  = mean(AR(t))，跨所有样本事件
累计平均超额 CAAR = Σ AAR(t)
```

- **R_stock(t)**：个股第 t 日涨跌幅（`daily.pctChg / 100`）
- **R_benchmark(t)**：基准指数第 t 日涨跌幅（默认沪深300 `000300.SH`，`indexDaily.pctChg / 100`）
- 事件窗口默认 `pre=5, post=20`，即事件日前 5 个交易日到后 20 个交易日

```typescript
@Injectable()
export class EventStudyService {
  constructor(private readonly prisma: PrismaService) {}

  /** 获取所有支持的事件类型 */
  getEventTypes(): EventTypeConfig[] {
    return Object.values(EVENT_TYPE_CONFIGS)
  }

  /** 分页查询事件样本列表 */
  async queryEvents(dto: EventStudyEventsQueryDto): Promise<{ total: number; items: any[] }> {
    // 根据 EventType 查询对应数据表
    // 使用 $queryRaw 带上 defaultFilter + 用户 tsCode/startDate/endDate 过滤
    // 分页返回
  }

  /** 核心：事件影响分析 */
  async analyze(dto: EventStudyAnalyzeDto): Promise<EventStudyResultDto> {
    const config = EVENT_TYPE_CONFIGS[dto.eventType]
    const preDays = dto.preDays ?? 5
    const postDays = dto.postDays ?? 20
    const benchmarkCode = dto.benchmarkCode ?? '000300.SH'
    const windowSize = preDays + 1 + postDays // 总窗口交易日数

    // Step 1: 提取事件样本
    const events = await this.extractEvents(config, dto)

    // Step 2: 逐事件计算 AR 序列
    const validSamples: EventSampleDto[] = []
    for (const event of events) {
      const sample = await this.computeEventAR(event.tsCode, event.eventDate, preDays, postDays, benchmarkCode)
      if (sample) validSamples.push(sample)
    }

    // Step 3: 聚合 AAR / CAAR
    const { aarSeries, caarSeries } = this.aggregateAAR(validSamples, windowSize)

    // Step 4: t 检验
    const { tStatistic, pValue } = this.tTest(validSamples, windowSize)

    // Step 5: 排序取 top/bottom
    const sorted = [...validSamples].sort((a, b) => b.car - a.car)

    return {
      eventType: dto.eventType,
      eventLabel: config.label,
      sampleCount: validSamples.length,
      window: `[-${preDays}, +${postDays}]`,
      benchmark: benchmarkCode,
      aarSeries,
      caarSeries,
      caar: caarSeries[caarSeries.length - 1] ?? 0,
      tStatistic,
      pValue,
      topSamples: sorted.slice(0, 10),
      bottomSamples: sorted.slice(-10).reverse(),
    }
  }
}
```

### 3.5 核心计算逻辑

#### 事件样本提取（SQL）

```sql
-- 以 FORECAST 为例
SELECT DISTINCT f.ts_code AS "tsCode", f.ann_date AS "eventDate"
FROM earnings_forecast_reports f
JOIN stock_basic_profiles sb ON f.ts_code = sb.ts_code AND sb.list_status = 'L'
WHERE f.ann_date >= :startDate
  AND f.ann_date <= :endDate
  -- 可选：AND f.ts_code = :tsCode
  -- defaultFilter 追加
ORDER BY f.ann_date DESC
LIMIT 2000
```

> **去重**：同一只股票在同一天公告多个事件时，仅保留一条（`DISTINCT ON (ts_code, event_date)`）。

#### 单事件 AR 序列计算

```sql
-- 获取事件日前后的交易日序列
WITH trade_days AS (
  SELECT trade_date,
         ROW_NUMBER() OVER (ORDER BY trade_date) AS rn
  FROM stock_daily_prices
  WHERE ts_code = :tsCode
    AND trade_date BETWEEN :windowStart AND :windowEnd
  ORDER BY trade_date
),
event_day AS (
  SELECT rn AS event_rn FROM trade_days
  WHERE trade_date = (
    -- 找到距离事件日最近的交易日（事件日可能非交易日）
    SELECT MIN(trade_date) FROM trade_days WHERE trade_date >= :eventDate
  )
)
SELECT
  td.trade_date,
  (td.rn - ed.event_rn) AS relative_day,
  d.pct_chg AS stock_return,
  id.pct_chg AS benchmark_return,
  (d.pct_chg - id.pct_chg) AS ar
FROM trade_days td
CROSS JOIN event_day ed
JOIN stock_daily_prices d ON d.ts_code = :tsCode AND d.trade_date = td.trade_date
LEFT JOIN index_daily_prices id ON id.ts_code = :benchmarkCode AND id.trade_date = td.trade_date
WHERE td.rn BETWEEN (ed.event_rn - :preDays) AND (ed.event_rn + :postDays)
ORDER BY td.trade_date
```

> **性能优化**：批量查询时先一次性拉取所有相关日期的日线数据到内存（按 `tsCode + dateRange`），在应用层计算 AR，避免逐事件发 SQL。

#### 伪代码：批量计算路径

```typescript
async computeEventAR(
  tsCode: string,
  eventDate: Date,
  preDays: number,
  postDays: number,
  benchmarkCode: string,
): Promise<EventSampleDto | null> {
  // 1. 查询 eventDate 附近的交易日列表（已缓存在内存 Map 中）
  const tradeDays = this.tradeDayCache.getRange(eventDate, -preDays - 10, postDays + 10)

  // 2. 定位 eventDate 在 tradeDays 中的位置（>=eventDate 的第一个交易日）
  const eventIdx = tradeDays.findIndex(d => d >= eventDate)
  if (eventIdx < preDays) return null // 前窗口数据不足

  // 3. 截取窗口
  const windowStart = eventIdx - preDays
  const windowEnd = eventIdx + postDays
  if (windowEnd >= tradeDays.length) return null // 后窗口数据不足

  // 4. 从预加载的行情 Map 中取 pctChg
  const arSeries: number[] = []
  for (let i = windowStart; i <= windowEnd; i++) {
    const date = tradeDays[i]
    const stockRet = this.stockReturns.get(`${tsCode}:${date}`) ?? 0
    const benchRet = this.benchReturns.get(`${benchmarkCode}:${date}`) ?? 0
    arSeries.push(stockRet - benchRet)
  }

  const car = arSeries.reduce((sum, ar) => sum + ar, 0)

  return {
    tsCode,
    name: this.stockNames.get(tsCode) ?? null,
    eventDate: formatDate(eventDate),
    car: round(car, 4),
    arSeries: arSeries.map(v => round(v, 4)),
  }
}
```

#### AAR / CAAR 聚合

```typescript
aggregateAAR(
  samples: EventSampleDto[],
  windowSize: number,
): { aarSeries: number[]; caarSeries: number[] } {
  const n = samples.length
  if (n === 0) return { aarSeries: [], caarSeries: [] }

  const aarSeries = new Array(windowSize).fill(0)

  for (const sample of samples) {
    for (let t = 0; t < windowSize && t < sample.arSeries.length; t++) {
      aarSeries[t] += sample.arSeries[t] / n
    }
  }

  const caarSeries: number[] = []
  let cumSum = 0
  for (const aar of aarSeries) {
    cumSum += aar
    caarSeries.push(round(cumSum, 4))
  }

  return {
    aarSeries: aarSeries.map(v => round(v, 4)),
    caarSeries,
  }
}
```

#### t 检验

```typescript
tTest(
  samples: EventSampleDto[],
  windowSize: number,
): { tStatistic: number; pValue: number } {
  const n = samples.length
  if (n < 2) return { tStatistic: 0, pValue: 1 }

  // 对最终 CAR 做单样本 t 检验（H0: CAR = 0）
  const cars = samples.map(s => s.car)
  const mean = cars.reduce((a, b) => a + b, 0) / n
  const variance = cars.reduce((sum, c) => sum + (c - mean) ** 2, 0) / (n - 1)
  const se = Math.sqrt(variance / n)
  const tStat = se > 0 ? mean / se : 0

  // 近似 p 值（双侧，使用正态近似，n 较大时足够准确）
  const pValue = 2 * (1 - normalCDF(Math.abs(tStat)))

  return { tStatistic: round(tStat, 4), pValue: round(pValue, 6) }
}

/** 标准正态分布 CDF 近似（Abramowitz & Stegun 26.2.17） */
function normalCDF(x: number): number {
  const a1 = 0.254829592, a2 = -0.284496736, a3 = 1.421413741
  const a4 = -1.453152027, a5 = 1.061405429, p = 0.3275911
  const sign = x < 0 ? -1 : 1
  x = Math.abs(x) / Math.SQRT2
  const t = 1.0 / (1.0 + p * x)
  const y = 1 - ((((a5 * t + a4) * t + a3) * t + a2) * t + a1) * t * Math.exp(-x * x)
  return 0.5 * (1 + sign * y)
}
```

```typescript
@Injectable()
export class EventSignalService {
  private readonly logger = new Logger(EventSignalService.name)

  constructor(
    private readonly prisma: PrismaService,
    private readonly eventsGateway: EventsGateway,
  ) {}

  // ── 规则 CRUD ─────────────────────────────────────────────────────

  async createRule(userId: number, dto: CreateSignalRuleDto): Promise<EventSignalRule> {
    return this.prisma.eventSignalRule.create({
      data: { userId, ...dto },
    })
  }

  async listRules(userId: number, page = 1, pageSize = 20) {
    const [items, total] = await Promise.all([
      this.prisma.eventSignalRule.findMany({
        where: { userId, status: { not: 'DELETED' } },
        orderBy: { createdAt: 'desc' },
        skip: (page - 1) * pageSize,
        take: pageSize,
      }),
      this.prisma.eventSignalRule.count({
        where: { userId, status: { not: 'DELETED' } },
      }),
    ])
    return { items, total, page, pageSize }
  }

  async updateRule(userId: number, ruleId: number, dto: UpdateSignalRuleDto) {
    return this.prisma.eventSignalRule.update({
      where: { id: ruleId, userId },
      data: dto,
    })
  }

  async deleteRule(userId: number, ruleId: number) {
    return this.prisma.eventSignalRule.update({
      where: { id: ruleId, userId },
      data: { status: 'DELETED' },
    })
  }

  // ── 信号扫描 ─────────────────────────────────────────────────────

  /**
   * 扫描当日新增事件，匹配所有 ACTIVE 规则生成信号。
   *
   * 调用时机：
   *   1. 每日 Tushare 数据同步完成后自动调用
   *   2. 管理端 POST /event-study/signal-rules/scan 手动触发
   *
   * 流程：
   *   1. 按 eventType 分组查询所有 ACTIVE 规则
   *   2. 对每种事件类型，查询 tradeDate 当日的新增事件
   *   3. 对每条事件 × 每条规则，评估 conditions 是否匹配
   *   4. 匹配则写入 EventSignal + 通过 WebSocket 推送给规则所属用户
   */
  async scanAndGenerate(tradeDate?: string): Promise<{ signalsGenerated: number }> {
    const dateStr = tradeDate ?? this.getLatestTradeDateStr()
    this.logger.log(`开始事件信号扫描，目标日期：${dateStr}`)

    const rules = await this.prisma.eventSignalRule.findMany({
      where: { status: 'ACTIVE' },
    })

    // 按 eventType 分组
    const rulesByType = new Map<string, EventSignalRule[]>()
    for (const rule of rules) {
      const list = rulesByType.get(rule.eventType) ?? []
      list.push(rule)
      rulesByType.set(rule.eventType, list)
    }

    let signalsGenerated = 0

    for (const [eventType, typeRules] of rulesByType) {
      const config = EVENT_TYPE_CONFIGS[eventType as EventType]
      if (!config) continue

      // 查询当日新增事件
      const events = await this.queryTodayEvents(config, dateStr)

      for (const event of events) {
        for (const rule of typeRules) {
          if (this.matchConditions(event, rule.conditions as Record<string, unknown>)) {
            await this.prisma.eventSignal.create({
              data: {
                ruleId: rule.id,
                tsCode: event.tsCode,
                stockName: event.name ?? null,
                eventDate: new Date(dateStr.replace(/(\d{4})(\d{2})(\d{2})/, '$1-$2-$3')),
                signalType: rule.signalType,
                eventDetail: event,
              },
            })
            signalsGenerated++

            // WebSocket 推送给规则所属用户
            this.eventsGateway.sendToUser(rule.userId, 'event-signal', {
              ruleId: rule.id,
              ruleName: rule.name,
              tsCode: event.tsCode,
              stockName: event.name,
              signalType: rule.signalType,
              eventDate: dateStr,
            })
          }
        }
      }
    }

    this.logger.log(`事件信号扫描完成，生成 ${signalsGenerated} 条信号`)
    return { signalsGenerated }
  }

  /**
   * 条件匹配器：将规则 conditions JSON 与事件记录字段做比较。
   *
   * 支持的操作符：
   *   - 直接值：{ "type": "预增" } → event.type === "预增"
   *   - gte/lte/gt/lt：{ "pChangeMin": { "gte": 50 } } → event.pChangeMin >= 50
   *   - in：{ "type": { "in": ["预增", "略增"] } }
   */
  private matchConditions(event: Record<string, unknown>, conditions: Record<string, unknown>): boolean {
    if (!conditions || Object.keys(conditions).length === 0) return true

    for (const [field, expected] of Object.entries(conditions)) {
      const actual = event[field]
      if (actual == null) return false

      if (typeof expected === 'object' && expected !== null && !Array.isArray(expected)) {
        const ops = expected as Record<string, unknown>
        if ('gte' in ops && (actual as number) < (ops.gte as number)) return false
        if ('lte' in ops && (actual as number) > (ops.lte as number)) return false
        if ('gt' in ops && (actual as number) <= (ops.gt as number)) return false
        if ('lt' in ops && (actual as number) >= (ops.lt as number)) return false
        if ('in' in ops && !(ops.in as unknown[]).includes(actual)) return false
      } else {
        if (actual !== expected) return false
      }
    }

    return true
  }
}
```
